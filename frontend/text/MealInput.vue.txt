<script setup>
import { ref, onUnmounted } from 'vue'
import { analyzeMeal } from '../api'
import { eventBus } from '../utils/eventBus'

const mealText = ref('')
const result = ref(null)
const loading = ref(false)
const showResult = ref(false)

const loadData = async () => {
  // MealInput doesn't need to load initial data
}

const handleAnalyze = async () => {
  if (!mealText.value.trim()) return
  
  loading.value = true
  const res = await analyzeMeal({
    user_id: 1,
    meal_text: mealText.value
  })
  result.value = res.data
  loading.value = false
  showResult.value = true
  
  eventBus.value.emit('meal:analyzed')
}

const getRiskClass = (level) => {
  return level || 'low'
}

const getRiskLabel = (level) => {
  const map = { high: '高风险', medium: '中等', low: '低风险' }
  return map[level] || level
}

defineExpose({ loadData })
</script>

<template>
  <div class="meal-input">
    <div class="card-header">
      <span class="icon">🍽️</span>
      <span class="title">饮食记录与分析</span>
    </div>

    <div class="input-area">
      <textarea 
        v-model="mealText" 
        placeholder="请输入您吃了什么，例如：早餐吃了两个馒头、一杯豆浆"
        rows="3"
      ></textarea>
      <button class="analyze-btn" @click="handleAnalyze" :disabled="loading">
        {{ loading ? '分析中...' : '分析' }}
      </button>
    </div>

    <div v-if="showResult && result" class="result-area">
      <div class="result-header">
        <span :class="['risk-level', getRiskClass(result.risk_level)]">
          {{ getRiskLabel(result.risk_level) }}
        </span>
        <span class="score">评分: {{ result.score }}</span>
      </div>

      <div v-if="result.detected_foods?.length" class="foods">
        <div class="foods-title">检测到:</div>
        <div class="food-tags">
          <span 
            v-for="food in result.detected_foods" 
            :key="food.name"
            :class="['food-tag', food.category]"
          >
            {{ food.name }}
          </span>
        </div>
      </div>

      <div v-if="result.suggestion?.length" class="suggestions">
        <div class="suggestions-title">建议:</div>
        <div v-for="(s, idx) in result.suggestion" :key="idx" class="suggestion-item">
          {{ s }}
        </div>
      </div>
    </div>
  </div>
</template>
