<script setup>
import { ref, onMounted, onUnmounted } from 'vue'
import { getAdherenceAnalysis } from '../api'
import { eventBus } from '../utils/eventBus'

const data = ref(null)
const loading = ref(true)
const todos = ref([])

const loadData = async () => {
  loading.value = true
  const res = await getAdherenceAnalysis()
  data.value = res.data
  loading.value = false
  
  if (data.value) {
    todos.value = generateTodos(data.value)
  }
}

const generateTodos = (adherenceData) => {
  const items = []
  
  if (adherenceData.risk_level === 'high' || adherenceData.risk_level === 'medium') {
    items.push({
      id: 'glucose',
      title: '测量血糖',
      desc: '请在餐前/餐后测量血糖',
      icon: '📊',
      urgent: adherenceData.risk_level === 'high'
    })
  }
  
  if (adherenceData.glucose_summary?.includes('高') || adherenceData.glucose_summary?.includes('异常')) {
    items.push({
      id: 'glucose-alert',
      title: '关注血糖',
      desc: '近期血糖有异常，请密切监测',
      icon: '⚠️',
      urgent: true
    })
  }
  
  items.push({
    id: 'meal',
    title: '记录饮食',
    desc: '记录今日饮食情况',
    icon: '🍽️',
    urgent: false
  })
  
  items.push({
    id: 'medication',
    title: '按时服药',
    desc: '遵医嘱服用药物',
    icon: '💊',
    urgent: false
  })
  
  return items
}

const getLevelLabel = (level) => {
  const map = { high: '高风险', medium: '中等', low: '良好' }
  return map[level] || level
}

const getLevelClass = (level) => {
  return level || 'low'
}

onMounted(loadData)

const handleRefresh = () => loadData()

eventBus.value.on('glucose:added', handleRefresh)
eventBus.value.on('meal:analyzed', handleRefresh)
eventBus.value.on('medication:confirmed', handleRefresh)
eventBus.value.on('medication:taken', handleRefresh)

onUnmounted(() => {
  eventBus.value.off('glucose:added', handleRefresh)
  eventBus.value.off('meal:analyzed', handleRefresh)
  eventBus.value.off('medication:confirmed', handleRefresh)
  eventBus.value.off('medication:taken', handleRefresh)
})

defineExpose({ loadData })
</script>

<template>
  <div class="adherence-card">
    <div class="card-header">
      <span class="icon">📋</span>
      <span class="title">今日待办</span>
      <button class="refresh-btn" @click="loadData" :disabled="loading">
        {{ loading ? '加载中...' : '刷新' }}
      </button>
    </div>

    <div v-if="loading" class="loading">加载中...</div>
    <div v-else-if="todos.length === 0" class="empty">暂无待办</div>
    <div v-else class="todos-list">
      <div v-for="todo in todos" :key="todo.id" :class="['todo-item', { urgent: todo.urgent }]">
        <span class="todo-icon">{{ todo.icon }}</span>
        <div class="todo-content">
          <div class="todo-title">{{ todo.title }}</div>
          <div class="todo-desc">{{ todo.desc }}</div>
        </div>
        <div v-if="todo.urgent" class="urgent-tag">紧急</div>
      </div>
    </div>
  </div>
</template>
